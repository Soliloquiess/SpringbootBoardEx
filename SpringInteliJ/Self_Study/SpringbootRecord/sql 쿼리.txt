 CREATE SEQUENCE  "SCOTT"."SQ_STUDY_RECORD"  
MINVALUE 1 
MAXVALUE 9999 
INCREMENT BY 1 
START WITH 1001 
NOCACHE  NOORDER  NOCYCLE  ;


create table study_record(
    key_id number(10) ,
    study_day varchar2(8),
    contents varchar2(100),
    reg_day date,
    CONSTRAINT pk_study_record PRIMARY KEY(key_id)
);
